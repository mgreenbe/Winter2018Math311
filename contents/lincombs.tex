\end{scope}

\begin{scope}[rotate=-40]

\coordinate (aPplusbQ) at (\a*\Px + \b*\Qx, \a*\Py + \b*\Qy);
\coordinate (aPplusbQplusR) at (\a*\Px + \b*\Qx + \Rx, \a*\Py + \b*\Qy + \Ry);
